persamaan kuadrat

x² - (m + n)x + mn = 0

(x - m)(x - n) = 0

a.

x² - 2x - 8 = 0

2 bilangan (m dan n) :

dikalikan (-8)

dijumlahkan (-2)

m = -4

n = 2

x² - 2x - 8 = 0

(x - 4)(x + 2) = 0

x = 4 atau x = -2

b.

2x² - x - 3 = 0

(2x - 3)(x + 1) = 0

2x - 3 = 0

2x = 3

x = 3/2

x + 1 = 0

x = -1

x = {-1 , 3/2}

c.

x² - 16 = 0

x² - 4² = 0

(x + 4)(x - 4) = 0

x = -4 atau x = 4

d.

x² + x - 2 = 0

(x + 2)(x - 1) = 0

x = -2 atau x = 1

e.

x² - 9x + 20 = 0

(x - 4)(x - 5) = 0

x = 4 atau x = 5
