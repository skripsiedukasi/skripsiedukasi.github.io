[tex] = 2 \times ( \frac{1}{2} \pi {28}^{2} - \frac{1}{2}\pi {21}^{2} + \frac{1}{2} \pi {7}^{2} ) \\ = 2 \times \frac{1}{2}\pi( {28}^{2} - {21}^{2} + {7}^{2} ) \\ = \frac{22}{7} (784 - 441 + 49) \\ = \frac{22}{7} (392) = 1232 \: {cm}^{2} [/tex]
